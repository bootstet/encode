#

# з®Җзӯ”йўҳ

# 1.з®Җиҝ° Node.js зҡ„зү№зӮ№д»ҘеҸҠйҖӮз”Ёзҡ„еңәжҷҜгҖӮ

  • зү№зӮ№пјҡ
    • ReactorжЁЎејҸпјҢеҚ•зәҝзЁӢе®ҢжҲҗеӨҡзәҝзЁӢе·ҘдҪңпјҲеҚ•зәҝзЁӢпјү
    • ReactorжЁЎејҸдёӢе®һзҺ°ејӮжӯҘIOгҖҒдәӢд»¶й©ұеҠЁпјҲдәӢд»¶й©ұеҠЁжһ¶жһ„пјү
    • NodejsйҖӮз”ЁIOеҜҶйӣҶеһӢй«ҳе№¶еҸ‘е№¶еҸ‘иҜ·жұӮ пјҲејӮжӯҘIOпјү
  • йҖӮз”ЁеңәжҷҜпјҡ
    • IO еҜҶйӣҶеһӢй«ҳе№¶еҸ‘иҜ·жұӮ пјҲNode.jsдҪңдёәдёӯй—ҙеұӮпјүжҸҗй«ҳеҗһеҗҗйҮҸпјҢеҫҲж–№дҫҝзҡ„еӨ„зҗҶж•°жҚ®
    • ж“ҚдҪңж•°жҚ®еә“жҸҗдҫӣ API жңҚеҠЎ
    • е®һж—¶иҒҠеӨ©еә”з”ЁзЁӢеәҸ
  • дҪңз”Ёпјҡ
    • е®һзҺ°иҪ»йҮҸзә§гҖҒй«ҳжҖ§иғҪеҸҜжү©еұ•зҡ„WebжңҚеҠЎ
    • еүҚеҗҺз«ҜJavaScriptеҗҢжһ„ејҖеҸ‘
    • е®һзҺ°дҫҝжҚ·й«ҳж•Ҳзҡ„еүҚз«Ҝе·ҘзЁӢеҢ–

# 2.з®Җиҝ° Buffer зҡ„дҪҝз”ЁгҖӮеҢ…жӢ¬еӨҡз§ҚеҲӣе»әж–№ејҸгҖӮе®һдҫӢж–№жі•пјҢйқҷжҖҒж–№жі•гҖӮ

  • еҲӣе»әBufferе®һдҫӢ
    • alloc: еҲӣе»әжҢҮе®ҡеӯ—иҠӮеӨ§е°Ҹзҡ„buffer
    • allocUnsafeпјҡеҲӣе»әжҢҮе®ҡз®ЎеӨ§е°Ҹзҡ„Buffer
    • fromпјҡжҺҘеҸ—ж•°жҚ®пјҢеҲӣе»әbuffer
  • Bufferе®һдҫӢж–№жі•
    • fillпјҡ дҪҝз”Ёж•°жҚ®еЎ«е…… buffer
      let buf = Buffer.alloc(6)
  buf.fill('123')
    • write: еҗ‘bufferдёӯеҶҷе…Ҙж•°жҚ®
    • toString: д»ҺbufferдёӯжҸҗеҸ–ж•°жҚ®
    • sliceпјҡ жҲӘеҸ–buffer
    • indexOfпјҡ еңЁbufferдёӯжҹҘжүҫж•°жҚ®
    • copyпјҡжӢ·иҙқbufferдёӯзҡ„ж•°жҚ®
  • Buffer йқҷжҖҒж–№жі•
    • concat: е°ҶеӨҡдёӘbufferжӢјжҺҘжҲҗдёҖдёӘж–°зҡ„buffer
    • isBufferпјҡ еҲӨж–ӯеҪ“еүҚиҜ—еҸҘжҳҜеҗҰдёә buffer

# 3.еҶҷеҮә5дёӘд»ҘдёҠж–Үд»¶ж“ҚдҪңзҡ„APIпјҢе№¶дё”з”Ёж–Үеӯ—иҜҙжҳҺе…¶еҠҹиғҪгҖӮ

  • appendFileпјҡиҝҪеҠ зҡ„ж–№ејҸжғіжҢҮе®ҡзҡ„ж–Үд»¶дёӯеҶҷе…Ҙж•°жҚ®
  • copyFileпјҡ е°ҶжҹҗдёӘж–Үд»¶дёӯзҡ„ж•°жҚ®жӢ·иҙқиҮіеҸҰдёҖж–Үд»¶

# 4.з®Җиҝ°дҪҝз”ЁжөҒж“ҚдҪңзҡ„дјҳеҠҝпјҢд»ҘеҸҠNodeдёӯжөҒзҡ„еҲҶзұ»гҖӮ

  • жҲ‘д»¬еңЁиҜ»еҸ–иө„жәҗж—¶пјҢйҮҮз”ЁreadFileзҡ„ж–№ејҸзҡ„ж—¶еҖҷејҖй”ҖиҫғеӨ§пјҢеӣ жӯӨйҮҮз”ЁжөҒж“ҚдҪңзҡ„дјҳеҠҝжңүпјҡ
    • ж—¶й—ҙж•ҲзҺҮпјҡ жөҒзҡ„еҲҶж®өеӨ„зҗҶеҸҜд»ҘеҗҢдәӢж“ҚдҪңеӨҡдёӘж•°жҚ®chunk
    • з©әй—ҙж•ҲзҺҮпјҡ еҗҢдёҖж—¶й—ҙжөҒж— йЎ»еҚ жҚ®еӨ§еҶ…еӯҳз©әй—ҙ
    • дҪҝз”Ёж–№дҫҝпјҡ жөҒй…ҚеҗҲз®ЎзҗҶпјҢжү©еұ•зЁӢеәҸеҸҳеҫ—з®ҖеҚ•
  • жөҒзҡ„еҲҶзұ»пјҡ
    • ReadbleпјҡеҸҜиҜ»жөҒпјҢиғҪеӨҹе®һзҺ°ж•°жҚ®зҡ„иҜ»еҸ–
    • Writeableпјҡ еҸҜеҶҷжөҒпјҢиғҪеӨҹе®һзҺ°ж•°жҚ®зҡ„еҶҷж“ҚдҪң
    • DuplexпјҡеҸҢе·ҘжөҒпјҢеҚіеҸҜиҜ»еҸҲеҸҜеҶҷ
    • Tranformпјҡ иҪ¬жҚўжөҒпјҢеҸҜиҜ»еҸҜеҶҷпјҢиҝҳиғҪе®һзҺ°ж•°жҚ®иҪ¬жҚў

# 5.еңЁж•°жҚ®е°ҒиЈ…дёҺи§Је°ҒиЈ…иҝҮзЁӢдёӯпјҢй’ҲеҜ№еә”з”ЁеұӮгҖҒдј иҫ“еұӮгҖҒзҪ‘з»ңеұӮгҖҒж•°жҚ®й“ҫи·ҜеұӮгҖҒзү©зҗҶеұӮ5еұӮеҲҶеҲ«еҒҡдәҶд»Җд№ҲдәӢжғ…пјҹ

  • ж•°жҚ®е°ҒиЈ…
    • еә”з”ЁеұӮпјҡ еңЁеә”з”ЁеұӮпјҢдә§еҮәйңҖиҰҒиҝҗиҫ“зҡ„ж•°жҚ® data
    • дј иҫ“еұӮпјҡ з«ҜеҸЈзҡ„дҪңз”Ёз”ЁдәҺеңЁдё»жңәдёҠзЎ®е®ҡдёҖдёӘдё»жңәзҡ„иҝӣзЁӢпјҢж•°жҚ®еңЁиҝҷеұӮдјҡиў«еҢ…иЈ№дёҠзӣ®ж Үеә”з”Ёз«ҜеҸЈе’Ңиҝҗз”ЁеңЁеҪ“еүҚдё»жңәдёҠзҡ„жәҗз«ҜеҸЈпјҢд№ӢеҗҺж•°жҚ®еңЁдј еҗ‘зҪ‘з»ңеұӮ
      зӣ®ж Үз«ҜеҸЈ + жәҗз«ҜеҸЈ + data
    • зҪ‘з»ңеұӮпјҡ еӣ дёәжҲ‘д»¬зҡ„дё»жңәйғҪеӨ„дәҺдёҚеҗҢзҡ„зҪ‘з»ңйҮҢпјҢжүҖд»ҘжҲ‘д»¬йңҖиҰҒйҖҡиҝҮipеҚҸи®®жқҘзЎ®е®ҡзӣ®ж Үзӣ®ж Үдё»жңәжүҖеӯҳеңЁзҡ„зҪ‘з»ңпјҢеӣ жӯӨж•°жҚ®еңЁиҝҷеұӮдёҠдјҡиў«еҢ…иЈ№дёҠзӣ®ж Үдё»жңәзҡ„ipең°еқҖе’Ңжәҗipең°еқҖгҖӮ зҺ°еңЁеҸҜд»ҘзЎ®е®ҡжҹҗдёҖдёӘзҪ‘з»ңпјҢе№¶дёҚи№ҰзЎ®е®ҡеңЁиҝҷдёӘзҪ‘з»ңйҮҢзҡ„йӮЈеҸ°жңәеҷЁдҪҝжҲ‘д»¬йңҖиҰҒзҡ„ зӣ®ж Үip + жәҗip + зӣ®ж Үз«ҜеҸЈ + жәҗз«ҜеҸЈ + data
    • ж•°жҚ®й“ҫи·ҜеұӮпјҡ йҖҡиҝҮmacең°еқҖе®ҢжҲҗеҜ»еқҖж“ҚдҪңпјҢжүҖд»ҘеңЁиҝҷеұӮдёҠж•°жҚ®дјҡиў«еҢ…иЈ№дёҠзӣ®ж Үдё»жңәзҡ„macең°еқҖдёҺеҪ“еүҚдё»жңәзҡ„macең°еқҖпјҢиҮіжӯӨдёҖдёӘе®Ңж•ҙдҝЎжҒҜзҡ„ж•°жҚ®е°ұе°ҒиЈ…OKдәҶ зӣ®ж ҮMac + жәҗmac + зӣ®ж Үip + жәҗip + зӣ®ж Үз«ҜеҸЈ + жәҗз«ҜеҸЈ + data
    • зү©зҗҶеұӮпјҡж•°жҚ®жңҖз»ҲдјҡиҪ¬жҚўдәҢиҝӣеҲ¶ж•°жҚ®пјҢиҖҢзҪ‘зәҝжҳҜдёҚиғҪиҜҶеҲ«дәҢиҝӣеҲ¶зҡ„пјҢз»ҸиҝҮзҪ‘еҚЎи°ғиҜ•жҲҗй«ҳдҪҺз”өеҺӢгҖӮ
  • и§Је°ҒиЈ…
    • зү©зҗҶеұӮпјҡ жңүдәҶд»ҘдёҠиҝҷдәӣж•°жҚ®д»ҘеҗҺпјҢйҖҡиҝҮи·Ҝз”ұеҷЁзҡ„зҪ‘з»ңеҲҶй…Қе’Ңдј иҫ“жҲӘжӯўзҡ„иҝҗиҫ“пјҢжңҖз»ҲеҲ°зӣ®ж Үдё»жңәзҡ„зҪ‘еҚЎ иҝҷдёӘж—¶еҖҷзӣ®ж Үдё»жңәзҪ‘еҚЎдјҡиҝӣиЎҢж•°жҚ®и§Ји°ғпјҢе°Ҷз”өеҺӢе…ҲеҸҳдёәдәҢиҝӣеҲ¶пјҢз„¶еҗҺеңЁеҗ‘дёҠеұӮдј йҖ’иҮій“ҫи·ҜеұӮ
    • ж•°жҚ®й“ҫи·ҜеұӮпјҡиҝҷдёӘж—¶еҖҷдјҡеҲҶжһҗдёҖдёӢзӣ®ж Үзҡ„macең°еқҖжҳҜдёҚжҳҜеҪ“еүҚдё»жңәзҡ„macең°еқҖпјҢеҰӮжһңжҳҜе°ұз»§з»ӯеҗ‘дёҠдј йҖ’иҮізҪ‘з»ңеұӮ

  /**
   * й“ҫиЎЁз»“жһ„  з”ЁжқҘеӯҳж”ҫж•°жҚ®
   * 01 node + head + null
   * 02 head --> null
   * 03 size
   * 04 next element
   * 
   * 05 еўһеҠ  еҲ йҷӨ дҝ®ж”№ жҹҘиҜў жё…з©әиҠӮзӮ№
   */

  class Node {
    constructor(element, next) {
      this.element = element
      this.next = next
    }
  }

  class LinkedList {
    constructor(head, size) {
      this.head = null
      this.size = 0
    }
    // з§Ғжңүж–№жі•
    _getNode(index) {
      if(index < 0 || index >= this.size) {
        throw new Error('и¶Ҡз•ҢдәҶ')
      }
      let currentNode = this.head
      for (let i = 0; i <  index; i++) {
        currentNode = currentNode.next
      }
      return currentNode
    }
    add(index, element) {
      if (arguments.length === 1) {
        element = index
        index = this.size
      }
      if (index < 0 || index > this.size) {
        throw new Error('cross the border')
      }
      if (index === 0) {
        let head = this.head // дҝқеӯҳеҺҹжңүheadзҡ„жҢҮеҗ‘
        this.head = new Node(element, head)
      } else {
        let prevNode = this._getNode(index - 1)
        prevNode.next = new Node(element, prevNode.next)
      }
      this.size++
    }
    // еҲ йҷӨ
    remove(index) {
      let rmNode = null
      // еҲӨж–ӯеҲ йҷӨзҡ„жҳҜејҖеӨҙ
      if (index === 0) {
        rmNode = this.head
        if (!rmNode) {
          return undefined
        }
        this.head = rmNode.next
      } else {
        let prevNode = this._getNode(index - 1)
        rmNode = prevNode.next
        prevNode.next = rmNode.next
      }
      this.size--
      return rmNode
    }
    // и®ҫзҪ®
    set(index, element) {
      let node = this._getNode(index)
      node.element = element
    }
    // жҹҘиҜў
    get(index) {
      return this._getNode(index )
    }
    // жё…з©ә
    clear() {
      this.head = null
      this.size = 0
    }
  }


  class Queue{
    constructor() {
      this.LinkedList = new LinkedList()
    }
    // еҲ—е…Ҙ
    enQueue(data) {
      this.LinkedList.add(data)
    }
    // еҲ—еҮә
    deQueue() {
      return this.LinkedList.remove(0)
    }
  }

  const q = new Queue()
  q.enQueue('node1')
  q.enQueue('node2')

  let a = q.deQueue()
  a = q.deQueue()
  a = q.deQueue()
